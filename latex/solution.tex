\section{Problem 1}
Let $\mu$ be an unsigned measure and $\nu$ a signed measure (respectively) on a
measurable space $(X, \mathcal A)$.

\begin{lemma}[Part (a)]
  If $\nu \perp \mu$ and $\nu \ll \mu$, then $\nu = 0$.
\end{lemma}
\begin{proof}
  Fix an $E \in \mathcal A$. Since $\nu \perp \mu$, there exist disjoint $A$,
  $B$ such that for all $E$, $\mu(E) = \mu(E \cap A)$ and $\nu(E) = \nu(E \cap
  B)$.

  First, suppose that $E \subset A$: then since $A \cap B = \emptyset$, 
  \[
  \nu(E) = \nu(E \cap B) = 0
  \]

  Now suppose that $E \subset A^c$: then
  \[
  \mu(E) = \mu(E \cap A) = 0
  \]
  Hence since $\nu \ll \mu$, $\mu(E) = 0$.

  The claim follows for arbitrary $E$ by decomposing $E = E_1 \cup E_2$ for $E_1
  \subset A$, $E_2 \subset A^c$ and applying addivity.
\end{proof}

\begin{lemma}[Part (b)]
  Suppose $\nu_1 \perp \mu$ and $\nu_2 \perp \mu$. Then for any $\alpha_1,
  \alpha_2 \in \R$, for the signed measure:
  \[
  \nu = \alpha_1 \nu_1 + \alpha_2 \nu_2
  \]
  we have $\nu \perp \mu$.
\end{lemma}
\begin{proof}
  Since $\nu_1 \perp \mu, \nu_2 \perp \mu$, there exist $A_1, A_2$ disjoint with
  $B_1, B_2$ such that for all $E \in \mathcal A$,
  \begin{align*}
    \nu_1(E) &= \nu_1(E \cap A_1) \\
    \nu_2(E) &= \nu_2(E \cap A_2) \\
    \mu(E) &= \mu(E \cap B_1) = \mu(E \cap B_2)
  \end{align*}
  Note that then:
  \[
  \mu(E) = \mu(E \cap (B_1 \cap B_2))
  \]
  and thus we can take $B = B_1 \cap B_2$ for:
  \[
  \mu(E) = \mu(E \cap B)
  \]
  We want to show that there exists an $A$ disjoint with $B$ such that:
  \[
  \nu(E) = \nu(E \cap A)
  \]

  Setting $A = A_1 \cup A_2$, we have:
  \begin{align*}
    \nu(E \cap A) &=
    \alpha_1 \nu_1(E \cap A) +
    \alpha_2 \nu_2(E \cap A) \\
    &= \alpha_1 \nu_1((E \cap A) \cap A_1) +
    \alpha_2 \nu_2((E \cap A) \cap A_2) \\
    &= \alpha_1 \nu_1(E \cap A_1) + \alpha_2 \nu_2(E \cap A_2) \\
    &= \alpha_1 \nu_1(E) + \alpha_2 \nu_2(E) \\
  \end{align*}
  and thus the claim is satisfied.
\end{proof}

\begin{lemma}[Part (c)]
  Suppose $\nu_1 \ll \mu$ and $\nu_2 \ll \mu$. Then for any $\alpha_1, \alpha_2
  \in \R$, for the signed measure:
  \[
  \nu = \alpha_1 \nu_1 + \alpha_2 \nu_2
  \]
  we have $\nu \ll \mu$.
\end{lemma}
\begin{proof}
  Suppose $\mu(E) = 0$. Then:
  \begin{align*}
    \nu(E) &=
    \alpha_1 \nu_1(E) + 
    \alpha_2 \nu_2(E) \\
    &= 0
  \end{align*}
  because $\nu_1(E) = \nu_2(E) = 0$ since $\nu_1, \nu_2 \ll \mu$.
\end{proof}

\section{Problem 2}
\begin{theorem}
  \[
  \lim_{n \to \infty}
  \int_{[0, \infty)} n^{-1} e^{-nx^2} \, d\lambda =0
    \]
\end{theorem}

\begin{proof}
  Observe that, for:
  \[
  g(x) = e^{-x^2}
  \]
  For all $x \geq 1$, $e^{x^2} \geq x^2$, hence $e^{-x^2} \leq x^{-2}$. Recall
  that the infinite sum $\sum_{n=1}^\infty \frac{1}{x^2}$ converges, and so
  $\int_1^\infty g < \infty$. Then on the range $0 \leq x < 1$, $e^{-x^2}$ is
  bounded by $1$, and so $g \in L^1$.
  \[
  f_n(x) = \frac{1}{n} \cdot \frac{1}{e^{nx^2}}
  \]
  we have $f_n \leq g$ for all $n$. Hence by the dominated convergence theorem,
  \begin{align*}
  \lim_{n \to \infty}
  \int_{[0, \infty)} n^{-1} e^{-nx^2} \, d\lambda
    &=
  \int_{[0, \infty)}
    \left(\lim_{n \to \infty}
    n^{-1} e^{-nx^2}\right) \, d\lambda \\
    &= \int_{[0, \infty)} 0 \, d\lambda \\
    &= 0
  \end{align*}
\end{proof}

\section{Problem 3}
\begin{theorem}
  Suppose $f \colon \R \to \R$ is bounded, measurable, and:
  \[
  \lim_{h \to 0} \int_0^1 \frac{|f(x + h) - f(x)|}{h} \, dx = 0
  \]
  Show that $f$ is a.e. constant on the interval $[0, 1]$.
\end{theorem}

\begin{proof}
  Since $\frac{|f(x + h) - f(x)|}{h}$ is nonnegative, it follows that for every
  subinterval $[a, b] \subset [0, 1]$, we have:
  \[
  \lim_{h \to 0} \int_a^b \frac{|f(x + h) - f(x)|}{h} \, dx = 0
  \]
  Thus since:
  \[
  \left| \int_a^b \frac{f(x + h) - f(x)}{h} \, dx \right| \leq 
  \int_a^b \frac{|f(x + h) - f(x)|}{h} \, dx
  \]
  We have, for all $0 \leq a \leq b \leq 1$,
  \[
  \lim_{h \to 0} \int_a^b \frac{f(x + h) - f(x)}{h} \, dx  = 0
  \]
  Then
  \begin{align*}
    \int_a^b \frac{f(x + h) - f(x)}{h} \, dx
    &=
    \frac{1}{h}\int_a^b \left(f(x + h) - f(x)\right) \, dx \\
    &=
    \frac{1}{h}\left(\int_a^b f(x + h) \, dx - \int_a^b f(x) \, dx\right) \\
    &= \frac{1}{h}\left(\int_a^b f(x + h) \, dx - \int_a^b f(x) \, dx\right) \\
    &= \frac{1}{h}\left(\int_{a+h}^{b+h} f(x) \, dx - \int_a^b f(x) \, dx\right) \\
    &= \frac{1}{h}\left(\int_{b}^{b+h} f(x) \, dx - \int_a^{a+h} f(x) \, dx\right) \\
  \end{align*}
  By the Lebesgue differentiation theorem (which holds since $f$ is bounded and
  $[0, 1]$ is a closed interval, and so $f \in L^1$), for a.e. $a, b$, we have:
  \[
  \lim_{h \to 0}
  \int_a^b \frac{f(x + h) - f(x)}{h} \, dx  = f(b) - f(a)
  \]
  Hence $f(b) = f(a)$ for a.e. $a, b$, i.e., $f$ is a.e. constant on $[0, 1]$.
\end{proof}

\section{Problem 4}
\begin{theorem}
  Suppose $f \in L^2$ and for any rational $a, b$ with $a < b$, we have:
  \[
  \int_{[a, b]} f(x) \, dx =0
  \]
  Then $f(x) = 0$ a.e.
\end{theorem}

\begin{proof}
  First, we show that
  \[
  \int_{[a, b]} f(x) \, dx =0
  \]
  for arbitrary $[a, b]$ (not merely rational). Suppose that $a \in Q$ but $b
  \not\in Q$. By density, for every $\delta > 0$, there exists a $b' \in \Q, b'
  < b$ such that $m([b', b]) < \delta$. Furthermore, we note that if $f \in
  L^2(\R^d)$, then $f \in L^1(E)$ for any compact subset of $\R^d$. Hence by
  additvitiy and the triangle inequality:
  \begin{align*}
    |\int_{[a, b]} f|
    &= |\int_{[a, b']} f + \int_{[b', b]} f| \\
    &\leq |\int_{[a, b']} f|  + |\int_{[b', b]} f| \\
    &\leq |\int_{[a, b']} f|  + \int_{[b', b]} |f|  \\
    &\leq \int_{[b', b]} |f|
  \end{align*}
  Now by absolute continuity of the integral, for every $\epsilon > 0$, there
  exists a $\delta' > 0$ such that $\int_{E} |f| < \epsilon$ whenever $m(E) <
  \delta'$, and hence $\int_{[b', b]} |f| \to 0$ as $\delta \to 0$. Applying the
  same argument to an $a' \in Q, a < a'$, we obtain the desired result for
  arbitrary endpoints. \checkmark

  Now for every $x \in \R$, and every closed ball $B$ containing $x$, we have:
  \[
  \frac{1}{m(B)} \int_B f = 0
  \]
  Hence by the Lebesgue differentiation theorem, $f(x) = 0$ for every $x$ in the
  Lebesgue set of $x$, i.e. almost everywhere.
\end{proof}

\begin{theorem}
  Recall that, if $K(x, y) \in L^2(\R^d \times \R^d)$, then the operator $T$
  defined by:
  \[
  Tf(x) = \int_{\R^d} K(x, y) f(y) \, dy
  \]
  defines a Hilbert-Schmidt operator. Prove that $T$ is a bounded operator from
  $L^2(\R^d)$ to $L^2(\R^d)$.
\end{theorem}

\begin{proof}
  By Cauchy-Schwarz, we have:
  \begin{align*}
    |T(f)(x)|^2 &=
    \left|\int K(x, y) f(y) \, dy\right|^2 \\
    &= \leq \left(
    \int |K(x, y)|^2 \, dy
    \right)
    \left(
    \int |f(y)|^2 \, dy
    \right)
  \end{align*}
  Thus:
  \[
  |T(f)(x)|^2 \leq \|f\|^2_2 \int |K(x, y)|^2 \, dy 
  \]
  Integrating with respect to $x$,
  \[
  \|T f\|^2_2 \leq
  \|K\|^2_{L^2} \|f\|^2_2 \Rightarrow
  \|T f\|_2 \leq \|K\|_{L^2} \|f\|_2
  \]
  Thus $T$ is bounded, with operator norm at most $\|K\|_L^2$.
\end{proof}
